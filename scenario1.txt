// Prevent inserting Account Record without pnone number.
Trigger AccountTrigger on Account(before insert)
{
 if(Trigger.new!=null)
{
 for(Account a: Trigger.new)
{
if(a.Phone==null)
{
 //a.addError('You cannot leave the Phone field as empty');
a.Phone.addError('You cannot leave the Phone field as empty');
}
}
}
}
